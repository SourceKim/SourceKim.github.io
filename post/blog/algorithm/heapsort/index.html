堆排序的时间复杂度 最好最坏都是 O（nlogn），是 不稳定 排序

大顶堆 & 小顶堆

有两种堆：

  • 大顶堆：一棵完全二叉树，所有的根节点都大于其子节点 （此时，这个堆的最大值在堆顶）

  • 小顶堆：一棵完全二叉树，所有的根节点都小于其子节点 （此时，这个堆的最小值在堆顶）

堆 与 数组

数组从前往后遍历，将其按照 从树 顶端往底部、从左往右 构造出一棵完全二叉树

堆的性质

  • 对于数组位置 i 的元素，其左右孩子对应数组的位置分别是：2i + 1 、2i + 2

  • 倒数第一层的最后一个节点的位置是 数组的长度 len 的一半 - 1：即 len / 2 - 1

堆调整算法

设有 start 参数，代表要调整的开始节点

设有 len 参数，代表数组的长度

  • 先用 tmp 保存一下要调整的节点的值

    int tmp = nums[start]

  • 遍历 int i 从 start 的左子节点（2 * start + 1）开始，对比 start 的左右节点，让 i 指向他们的较大值

    for (int i = 2 * start + 1; i < len; i = 2 * i + 1) {

    if (i + 1 < len && nums[i] < nums[i + 1]) i++;

    .....
}

    为了拿到较大子节点的子节点，所以 i = 2 * i + 1 来更新 i，这样才能对比完子节点再对比孙节点

    这么做是为了 因为本次的交换，可能会影响到孙节点的堆结构

  • 如果当前的 i 节点的值 大于 tmp，则将 i 节点的值赋值给 start 节点 & 将 start = i；否则认为排序已经完成，直接 return

    for (int i = 2 * start + 1; i < len; i = 2 * i + 1) {

    if (i + 1 < len && nums[i] < nums[i + 1]) i++;

    if (nums[i] > tmp) {
        nums[start] = nums[i];
        start = i;
    } else {
        break;
    }
}

  • 最后将当前 tmp 的值给到 start 位置

    for () { ...... }

nums[start] = tmp

堆排序

主要分两步：

    1. 构造堆

    从 倒数第一层的最后一个节点， 即 len / 2 - 1 开始，一直遍历都 0，然后循环执行 堆调整

    此时会自自底向上地调整堆，最终得到 初始的大（小）顶堆

    1. 不断循环，将堆中的顶（即数组中最大或者最小值，位置是数组的第 0 个），放到数组末尾

C++ 代码

#include <iostream>
#include <vector>

using namespace std;

class HeapSorting {
public:
    
    /// 升序
    /// @param nums 要处理的数组
    void sort_ascend(vector<int> &nums) {
        
        int len = (int)nums.size();
        
        // 1. 先构造大顶堆
        for (int i = len / 2 - 1; i >= 0; i--) {
            adjust_big(nums, i, len);
        }
        
        // 2. 调整
        for (int i = len - 1; i > 0; i--) {
            swap(nums[0], nums[i]); // 交换栈顶和后面的元素
            adjust_big(nums, 0, i); // 重新调整
        }
    }
    
    /// 降序
    /// @param nums 要处理的数组
    void sort_descend(vector<int> &nums) {
        
        int len = (int)nums.size();
        
        // 1. 先构造小顶堆
        for (int i = len / 2 - 1; i >= 0; i--) {
            adjust_small(nums, i, len);
        }
        
        // 2. 调整
        for (int i = len - 1; i > 0; i--) {
            swap(nums[0], nums[i]); // 交换栈顶和后面的元素
            adjust_small(nums, 0, i); // 重新调整
        }
    }
    
    
private:
    
    /// 将给定的数组，从 idx 节点开始、自顶向下地调整堆，使堆成为 大顶堆
    /// @param nums 数组
    /// @param start 调整节点开始的位置
    /// @param length 调整节点结束的位置
    void adjust_big(vector<int> &nums, int start, int length) {
        
        int tmp = nums[start]; // 取出当前要判别的元素
        
        // i 表示每层的左子节点
        for (int i = 2 * start + 1; i < length; i = 2 * i + 1) {
            
            // 让 i 指向 左右子节点较大的那一个
            if (i + 1 < length && nums[i] < nums[i + 1]) {
                i++;
            }
            
            // 判断：较大的子节点是否大于 tmp，是则将子节点赋值给父节点
            if (nums[i] > tmp) {
                nums[start] = nums[i];
                start = i;
            } else {
                break;
            }
            
        }
        
        // 将这个值放到合适的位置
        nums[start] = tmp;
    }
    
    /// 将给定的数组，从 idx 节点开始、自顶向下地调整堆，使堆成为 小顶堆
    /// @param nums 数组
    /// @param start 调整节点开始的位置
    /// @param length 调整节点结束的位置
    void adjust_small(vector<int> &nums, int start, int length) {
        
        int tmp = nums[start]; // 取出当前要判别的元素
        
        // i 表示每层的左子节点
        for (int i = 2 * start + 1; i < length; i = 2 * i + 1) {
            
            // 让 i 指向 左右子节点较小的那一个
            if (i + 1 < length && nums[i] > nums[i + 1]) {
                i++;
            }
            
            // 判断：较小的子节点是否小于 tmp，是则将子节点赋值给父节点
            if (nums[i] < tmp) {
                nums[start] = nums[i];
                start = i;
            } else {
                break;
            }
            
        }
        
        // 将这个值放到合适的位置
        nums[start] = tmp;
    }
    
    
};

int main(int argc, const char * argv[]) {
    
    HeapSorting *sort = new HeapSorting();
    
    vector<int> data { 9, 8, 3, 7, 1, 10, 2, 5 };
    
    sort->sort_ascend(data);
    
    for (int d : data) {
        cout << d << " ";
    }
    
    return 0;
}
